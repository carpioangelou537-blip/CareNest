 table public.appointments (
  id uuid not null default extensions.uuid_generate_v4 (),
  title text not null,
  patient_id uuid not null,
  caregiver_id uuid null,
  date date not null,
  time time without time zone not null,
  duration_mins integer null default 60,
  location text null,
  notes text null,
  status text not null default 'scheduled'::text,
  reminder_sent boolean not null default false,
  created_by uuid null,
  created_at timestamp with time zone not null default now(),
  updated_at timestamp with time zone not null default now(),
  constraint appointments_pkey primary key (id),
  constraint appointments_caregiver_id_fkey foreign KEY (caregiver_id) references profiles (id) on delete set null,
  constraint appointments_created_by_fkey foreign KEY (created_by) references profiles (id) on delete set null,
  constraint appointments_patient_id_fkey foreign KEY (patient_id) references patients (id) on delete CASCADE,
  constraint appointments_status_check check (
    (
      status = any (
        array[
          'scheduled'::text,
          'in_progress'::text,
          'completed'::text,
          'cancelled'::text,
          'no_show'::text
        ]
      )
    )
  )
) TABLESPACE pg_default;

 table public.billings (
  id uuid not null default extensions.uuid_generate_v4 (),
  patient_id uuid not null,
  issued_by uuid null,
  description text not null,
  amount numeric(10, 2) not null,
  discount numeric(10, 2) null default 0,
  tax numeric(10, 2) null default 0,
  total numeric GENERATED ALWAYS as (((amount - discount) + tax)) STORED (10, 2) null,
  status text not null default 'pending'::text,
  payment_method text null,
  due_date date null,
  paid_at timestamp with time zone null,
  notes text null,
  created_at timestamp with time zone not null default now(),
  updated_at timestamp with time zone not null default now(),
  constraint billings_pkey primary key (id),
  constraint billings_patient_id_fkey foreign KEY (patient_id) references patients (id) on delete CASCADE,
  constraint billings_issued_by_fkey foreign KEY (issued_by) references profiles (id) on delete set null,
  constraint billings_status_check check (
    (
      status = any (
        array[
          'pending'::text,
          'paid'::text,
          'overdue'::text,
          'cancelled'::text,
          'refunded'::text
        ]
      )
    )
  ),
  constraint billings_amount_check check ((amount >= (0)::numeric)),
  constraint billings_tax_check check ((tax >= (0)::numeric)),
  constraint billings_discount_check check ((discount >= (0)::numeric)),
  constraint billings_payment_method_check check (
    (
      payment_method = any (
        array[
          'cash'::text,
          'gcash'::text,
          'maya'::text,
          'bank_transfer'::text,
          'insurance'::text,
          'other'::text
        ]
      )
    )
  )
) TABLESPACE pg_default;

 index IF not exists idx_billings_patient_id on public.billings using btree (patient_id) TABLESPACE pg_default;

 index IF not exists idx_billings_status on public.billings using btree (status) TABLESPACE pg_default;

 index IF not exists idx_billings_due_date on public.billings using btree (due_date) TABLESPACE pg_default;

  id uuid not null default extensions.uuid_generate_v4 (),
  full_name text not null,
  age integer null,
  gender text null,
  diagnosis text not null,
  medical_history text null,
  allergies text null,
  current_medications text null,
  status text not null default 'active'::text,
  assigned_caregiver uuid null,
  family_contact text null,
  emergency_contact text null,
  address text null,
  date_of_birth date null,
  blood_type text null,
  created_by uuid null,
  created_at timestamp with time zone not null default now(),
  updated_at timestamp with time zone not null default now(),
  constraint patients_pkey primary key (id),
  constraint patients_assigned_caregiver_fkey foreign KEY (assigned_caregiver) references profiles (id) on delete set null,
  constraint patients_created_by_fkey foreign KEY (created_by) references profiles (id) on delete set null,
  constraint patients_age_check check (
    (
      (age >= 0)
      and (age <= 150)
    )
  ),
  constraint patients_gender_check check (
    (
      gender = any (
        array['male'::text, 'female'::text, 'other'::text]
      )
    )
  ),
  constraint patients_blood_type_check check (
    (
      blood_type = any (
        array[
          'A+'::text,
          'A-'::text,
          'B+'::text,
          'B-'::text,
          'AB+'::text,
          'AB-'::text,
          'O+'::text,
          'O-'::text
        ]
      )
    )
  ),
  constraint patients_status_check check (
    (
      status = any (
        array[
          'active'::text,
          'stable'::text,
          'critical'::text,
          'discharged'::text
        ]
      )
    )
  )
